\subsection*{Seksjon 3.2}
\begin{enumerate}
    \item [\boxed{27}]
          \begin{enumerate}
              \item $n \log(n^2 + 1) + n^2 \log n \rightarrow O(n^2 \log n)$
              \item $(n \log n + 1)^2 + (\log n + 1)(n^2 + 1) \rightarrow O(n^2)$
          \end{enumerate}
    \item [\boxed{30}]
          \begin{enumerate}
              \item [(c)] \begin{flalign*}
                        \left\lfloor x + {1\over 2} \right\rfloor &\rightarrow O(n) &&\\
                        x &\rightarrow O(n) &&\\
                    \end{flalign*}
              \item [(e)] \begin{flalign*}
                        \log_{10}x &\rightarrow O(\log n) &&\\
                        \log_{2}x &\rightarrow O(\log n) &&\\
                    \end{flalign*}
          \end{enumerate}
\end{enumerate}

\subsection*{Seksjon 4.1}
\begin{enumerate}
    \item [\boxed{15}] \begin{enumerate}
              \item $(80 + 11) = 12 \cdot 7 + 7$, viseren står på 7, 80 timer etter den var 11
              \item $40 = 12 \cdot 3 + 4$, viseren sto på 8, 40 timer før den var 12
              \item $(100 + 6) = 12 \cdot 8 + 10$, viseren står på 10, 100 timer etter den var 6
          \end{enumerate}
    \item [\boxed{42}] \begin{flalign*}
              a \equiv b \Mod{m} & \Leftrightarrow (a - b) | m &&\\
              c \cdot (a - b) | m \cdot c &\Rightarrow (ac - bc) | mc &&\\
              (ac - bc) | mc & \Rightarrow \boxed{ac \equiv bc \Mod{mc}} &&\\
          \end{flalign*}
\end{enumerate}
