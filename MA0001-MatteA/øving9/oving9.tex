              f'(x)                                       & = \frac{-x+1}{e^x}                                          \\
              \frac{-x+1}{e^x}                            & = \frac{-x}{e^x} + \frac{1}{e^x}                            \\
              \text{Finner null}                          & \text{punktet til } f'(x)                                   \\
              \frac{-x}{e^x} + \frac{1}{e^x}              & = 0                                                         \\
              \frac{x}{e^x}                               & = \frac{1}{e^x}                                             \\
              x                                           & = 1                                                         \\
              f(1)                                        & = \frac{1}{e}                                               \\
              \text{Funksjonen } f                        & \text{ har ett ekstremalpunkt} = \left(1, {1\over e}\right) \\
          \end{align*}
